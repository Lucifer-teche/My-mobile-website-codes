  <script>
    function locomotive() {
      gsap.registerPlugin(ScrollTrigger);

      const locoScroll = new LocomotiveScroll({
        el: document.querySelector("#main"),
        smooth: true,
      });

      locoScroll.on("scroll", ScrollTrigger.update);

      ScrollTrigger.scrollerProxy("#main", {
        scrollTop(value) {
          return arguments.length
            ? locoScroll.scrollTo(value, 0, 0)
            : locoScroll.scroll.instance.scroll.y;
        },
        getBoundingClientRect() {
          return {
            top: 0,
            left: 0,
            width: window.innerWidth,
            height: window.innerHeight,
          };
        },
        pinType: document.querySelector("#main").style.transform
          ? "transform"
          : "fixed",
      });

      ScrollTrigger.addEventListener("refresh", () => locoScroll.update());
      ScrollTrigger.refresh();
    }
    locomotive();

    // Canvas setup
    const canvas = document.getElementById("background-canvas");
    const context = canvas.getContext("2d");

    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;

    window.addEventListener("resize", () => {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;
      render();
    });

    // Generate image paths dynamically
    function files(index) {
      const number = String(index + 1).padStart(4, "0");
      return `./images/male${number}.png`;
    }

    const frameCount = 300;
    const images = [];
    const imageSeq = { frame: 0 };

    // Preload all 300 images
    for (let i = 0; i < frameCount; i++) {
      const img = new Image();
      img.src = files(i);
      img.onerror = () => console.warn(`Image failed to load: ${img.src}`);
      images.push(img);
    }

    // Animate image sequence on scroll
    gsap.to(imageSeq, {
      frame: frameCount - 1,
      snap: "frame",
      ease: "none",
      scrollTrigger: {
        scrub: 0.15,
        trigger: "#page>canvas",
        start: "top top",
        end: "600% top",
        scroller: "#main",
      },
      onUpdate: render,
    });

    // Initial render after first image loads
    images[0].onload = render;

    function render() {
      scaleImage(images[imageSeq.frame], context);
    }

    function scaleImage(img, ctx) {
      const canvas = ctx.canvas;
      const hRatio = canvas.width / img.width;
      const vRatio = canvas.height / img.height;
      const ratio = Math.max(hRatio, vRatio);
      const centerShift_x = (canvas.width - img.width * ratio) / 2;
      const centerShift_y = (canvas.height - img.height * ratio) / 2;
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      ctx.drawImage(
        img,
        0,
        0,
        img.width,
        img.height,
        centerShift_x,
        centerShift_y,
        img.width * ratio,
        img.height * ratio
      );
    }

    // Pin canvas section
    ScrollTrigger.create({
      trigger: "#page>canvas",
      pin: true,
      scroller: "#main",
      start: "top top",
      end: "600% top",
    });

    // Pin each page section
    ["#page1", "#page2", "#page3"].forEach((id) => {
      gsap.to(id, {
        scrollTrigger: {
          trigger: id,
          start: "top top",
          end: "bottom top",
          pin: true,
          scroller: "#main",
        },
      });
    });

    // GitHub button click
    const gitHubButton = document.getElementById("github");
    gitHubButton.onclick = () => {
      window.open("https://github.com/harshitv54", "_self");
    };
  </script>
